        function this=Canopy(initMoist,TSurf,strCanopyTable)
            global CH_ICE CH_WATER
            %this.TSurf is the temperature of the foliage
            this=this@Medium(CH_WATER,CH_ICE,initMoist,TSurf);
            this.ReadCanopyAttribute(this,strCanopyTable);
        end
        function InitializeByZero(this,basin)
            %basin is the basin variable
            nCells=sum(basin.isOverstory(basin,basinMask));
            this.intRain=zeros(nCells,1);
            this.intSnow=zeros(nCells,1);
        end
